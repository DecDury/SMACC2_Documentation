59 template <typename TOrthogonal, typename TBehavior, typename... Args>
60 std::shared_ptr<TBehavior> ISmaccState::configure(Args &&... args)
61 {
62  std::string orthogonalkey = demangledTypeName<TOrthogonal>();
63  RCLCPP_INFO(
64  getLogger(), "[%s] Configuring orthogonal: %s", THIS_STATE_NAME, orthogonalkey.c_str());
65 
66  TOrthogonal * orthogonal = this->getOrthogonal<TOrthogonal>();
67  if (orthogonal != nullptr)
68  {
69  auto clientBehavior = std::shared_ptr<TBehavior>(new TBehavior(args...));
70  clientBehavior->currentState = this;
71  orthogonal->addClientBehavior(clientBehavior);
72  clientBehavior->template onOrthogonalAllocation<TOrthogonal, TBehavior>();
73  return clientBehavior;
74  }
75  else
76  {
77  RCLCPP_ERROR(
78  getLogger(), "[%s] Skipping client behavior creation in orthogonal [%s]. It does not exist.",
79  THIS_STATE_NAME, orthogonalkey.c_str());
80  return nullptr;
81  }
82 }

92 template <typename SmaccClientType>
93 void ISmaccState::requiresClient(SmaccClientType *& storage)
94 {
95  const char * sname = (demangleSymbol(typeid(*this).name()).c_str());
96  storage = nullptr;
97  auto & orthogonals = this->getStateMachine().getOrthogonals();
98  for (auto & ortho : orthogonals)
99  {
100  ortho.second->requiresClient(storage);
101  if (storage != nullptr) return;
102  }
103 
104  RCLCPP_ERROR(
105  getLogger(),
106  "[%s] Client of type '%s' not found in any orthogonal of the current state machine. This may "
107  "produce a segmentation fault if the returned reference is used.",
108  sname, demangleSymbol<SmaccClientType>().c_str());
109 }
